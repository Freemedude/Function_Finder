98bool import_file(const std::filesystem::path &path, std::vector<Function_Decl> &inout_functions,
99 std::string_view search_term)
100{
101 std::string content_str = read_file_to_string(path);
102 std::string_view content_view = content_str;
103
104 // Loop over every line in the file, checking to see if it starts with the search term.
105 for (size_t line = 1;; line++)
106 {
107 if (content_view.starts_with(search_term))
108 {
109 Function_Decl func;
110 func.line = line + 1;
111 func.file = path.generic_string();
112
113 // Skip the search term
114 auto func_view = skip_whitespace(content_view);
115 func_view = advance(func_view, search_term.size());
116 func_view = skip_whitespace(func_view);
117
118 bool success = import_function(func_view, &func);
119 if (!success)
120 {
121 return false;
122 }
123
124 inout_functions.push_back(func);
125 }
126
127
128 // Check for end of file.
129 auto next_line_end = content_view.find('\n');
130 if (next_line_end == std::string_view::npos)
131 {
132 break;
133 }
134 content_view = content_view.substr(next_line_end + 1);
135 }
136
137 return true;
138}

140bool import_function(std::string_view source, Function_Decl *out_function)
141{
142 // We got a comment!
143 if (source[0] == '/')
144 {
145 if (source[1] == '/') // Line comment. Nice and easy.
146 {
147 std::string_view note_view = advance(source, (size_t)2);
148 note_view = skip_whitespace(note_view);
149 auto line_end = note_view.find('\n');
150 out_function->note = std::string(note_view.begin(), note_view.begin() + line_end);
151 source = advance(source, (size_t)source.find('\n') + 1);
152 }
153 else if (source[1] == '*') // Block comment.
154 {
155 // Find the end of the comment.
156 auto comment_end = source.find("*/");
157 if (comment_end == std::string_view::npos)
158 {
159 // There is no end to this block comment. It will consume us all.
160 return false;
161 }
162
163 std::string_view note_view(source.begin() + 2, source.begin() + comment_end);
164 note_view = skip_whitespace(note_view);
165 out_function->note = std::string(note_view.begin(), note_view.end());
166
167 // Sanitize.
168 out_function->note = std::regex_replace(out_function->note, std::regex("\n"), "\\n");
169 source = advance(source, comment_end + 2); // +2 to skip the '*/' bit
170 }
171 }
172
173 // Check for 'inline' modifier
174 source = skip_whitespace(source);
175 if (source.starts_with("inline"))
176 {
177 source = advance(source, sizeof("inline"));
178 }
179
180 size_t final_length = get_type(source, out_function->return_type);
181 source = advance(source, final_length);
182 source = skip_whitespace(source);
183
184 size_t length = get_symbol(source, out_function->name);
185 if (!length)
186 {
187 std::cerr << "[ERROR] Failed to get function name '" << source << "'\n";
188 return false;
189 }
190
191 final_length += length;
192 source = advance(source, length);
193
194 bool success = get_arguments(source, out_function->arguments);
195 if (!success)
196 {
197 std::cerr << "[ERROR] Failed to get arguments for function '" << out_function->name << "'\n";
198 return false;
199 }
200
201 out_function->num_optional_args = (int)std::count_if(out_function->arguments.begin(), out_function->arguments.end(), [](Argument &arg)
202 { return arg.has_default_value; });
203 out_function->num_required_args = out_function->arguments.size() - out_function->num_optional_args;
204
205 return true;
206}

209bool export_functions(const Settings &settings, const std::vector<Function_Decl> &functions)
210{

218 return false;

220
221 Cpp_File_Writer w(file);
222

234size_t get_type(std::string_view source, Value_Type &out_type)
235{
236 size_t result = 0;
237 source = skip_whitespace(source);
238 std::string str;
239 size_t length = get_symbol(source, str);
240 if (!length)
241 {
242 std::cout << "Failed to get type string from '" << source << "'\n";
243 out_type = Value_Type::UNKNOWN;
244 return 0;
245 }
246
247 result += length;
248 source = advance(source, length);
249
250 Value_Type type = Value_Type::UNKNOWN;
251 if (str == "double")
252 {
253 type = Value_Type::DOUBLE;
254 }
255 if (str == "float")
256 {
257 type = Value_Type::FLOAT;
258 }
259 if (str == "int")
260 {
261 type = Value_Type::INTEGER;
262 }
263 else if (str == "std::string")
264 {
265 type = Value_Type::STRING;
266 }
267 else if (str == "bool")
268 {
269 type = Value_Type::BOOLEAN;
270 }
271 else if (str == "void")
272 {
273 type = Value_Type::VOID;
274 }
275
276 out_type = type;
277
278 return result;
279}

281size_t get_argument(std::string_view source, Argument &out_arg)
282{
283 source = skip_whitespace(source);
284 Value_Type type;
285 size_t final_length = get_type(source, type);
286 source = advance(source, final_length);
287 source = skip_whitespace(source);
288 std::string name;
289 size_t length = get_symbol(source, name);
290 if (!length)
291 {
292 std::cout << std::format("Failed to get argument name at position '{}'\n", source.data());
293 return 0;
294 }
295
296 final_length += length;
297 source = advance(source, length);
298
299 source = skip_whitespace(source);
300
301 out_arg.name = name;
302 out_arg.type = type;
303
304 // Check for default value
305 if (source.size() > 0 && source[0] == '=')
306 {
307 out_arg.has_default_value = true;
308 out_arg.default_value.type = type;
309 source = advance(source, (size_t)1);
310 source = skip_whitespace(source);
311 switch (type)
312 {
313 case Value_Type::BOOLEAN:
314 {
315
316 bool result;
317 size_t length = get_bool(source, result);
318 if (length)
319 {
320 out_arg.default_value.data.bool_value = result;
321 source = advance(source, length);
322 final_length += length;
323 }
324 else
325 {
326 return false;
327 }
328 break;
329 }
330 case Value_Type::FLOAT:
331 {
332 float result;
333 size_t length = get_float(source, result);
334 if (length)
335 {
336 out_arg.default_value.data.float_value = result;
337 source = advance(source, length);
338 final_length += length;
339 }
340 else
341 {
342 return 0;
343 }
344 break;
345 }
346 case Value_Type::DOUBLE:
347 {
348 double result;
349 size_t length = get_double(source, result);
350 if (length)
351 {
352 out_arg.default_value.data.double_value = result;
353 source = advance(source, length);
354 final_length += length;
355 }
356 else
357 {
358 return 0;
359 }
360 break;
361 }
362 case Value_Type::INTEGER:
363 {
364
365 int result;
366 size_t length = get_int(source, result);
367 if (length)
368 {
369 out_arg.default_value.data.int_value = result;
370 source = advance(source, length);
371 final_length += length;
372 }
373 else
374 {
375 return false;
376 }
377 break;
378 }
379 case Value_Type::STRING:
380 {
381 std::string result;
382 size_t length = get_quoted_string(source, result);
383 if (length)
384 {
385 strncpy(out_arg.default_value.data.string_value, result.c_str(), result.length());
386 source = advance(source, length);
387 final_length += length;
388 }
389 else
390 {
391 std::cout << "Failed to get string default value " << source << "\n";
392 return 0;
393 }
394 break;
395 }
396 }
397 }
398
399 return final_length;
400}

402bool get_arguments(std::string_view source, std::vector<Argument> &out_args)
403{
404 source = skip_whitespace(source);
405
406 if (source[0] != '(')
407 {
408 return false;
409 }
410 source = advance(source, (size_t)1);
411
412 // We should now be within the parenthesis
413 source = skip_whitespace(source);
414
415 // Loop over all args
416 while (true)
417 {
418 bool found_end_parenthesis = false;
419 size_t length = 0;
420
421 // Delimit each argument
422 while (true)
423 {
424 if (source[length] == ',')
425 {
426 length++;
427 break;
428 }
429
430 if (source[length] == ')')
431 {
432 found_end_parenthesis = true;
433 break;
434 }
435
436 length += 1;
437 }
438
439 if (length > 0)
440 {
441 Argument arg{};
442 bool success = get_argument(source.substr(0, length), arg);
443
444 if (!success)
445 {
446 return false;
447 }
448
449 out_args.push_back(arg);
450 }
451
452 if (found_end_parenthesis)
453 {
454 break;
455 }
456 source = advance(source, length);
457 }
458
459 return true;
460}

488void export_pre_declarations(Cpp_File_Writer &w, const std::vector<Function_Decl> &functions)
489{
490 // Write a section header to make it easier to navigate the output file.
491 w << "//////////////////////////////";
492 w << "// PRE-DECLARATIONS //";
493 w << "//////////////////////////////";
494
495 w << "// Pre-declarations";
496 w << "// Default values are ommitted because it: 1) wont compile with them defined twice; 2)"
497 "they are handled by the generated wrappers below.";
498 w.skip_line();
499 for (const auto &f : functions)
500 {
501 w << std::format("// From \"{}\" L{}", f.file, f.line);
502 auto &s = w.get_stream() << value_type_to_cpp_type(f.return_type) << " "
503 << f.name << "(";
504 for (int i = 0; i < f.arguments.size(); i++)
505 {
506 const Argument &arg = f.arguments[i];
507 s << value_type_to_cpp_type(arg.type) << " " << arg.name;
508
509 // We intentionally leave out the default value here. Since the compiler will
510 // complain if we define it twice. Default arguments are handled in the generated
511 // _wrapper_* functions placed below. @Daniel 10/04/2023
512 if (i != f.arguments.size() - 1)
513 {
514 s << ", ";
515 }
516 }
517 s << ");" << std::endl;
518 w.skip_line();
519 }

537void export_wrapper_function(Cpp_File_Writer &w, const Function_Decl &f)
538{
539 // Write the function definition
540 w << std::format("// Generated based on function \"{}\" from file \"{}\" L{}",
541 f.name, f.file, f.line);
542 w << std::format("inline bool _wrapper_{}(std::vector<std::string> &args, Value &out_result)",
543 f.name);
544 w << "{";
545 w.indent();
546
547 // Write the check to confirm that all needed arguments are provided
548 if (f.num_required_args > 0)
549 {
550 w << "// Check that all required arguments are provided.";
551 w << std::format("if(args.size() < {})", f.num_required_args);
552 w << "{";
553 w.indent();
554
555 w << std::format("printf(\"Not enough arguments for '{}'. Needed {}, but got %zu\\n\", "
556 "args.size());",
557 f.name, f.num_required_args);
558 w << "return false;";
559
560 w.unindent();
561 w << "}";
562 w.skip_line();
563 }
564
565 // Write the 'success' flag. This flag is reused to confirm all arguments are parsed correctly.
566 if (f.arguments.size() > 0)
567 {
568 w << "// Reused success flag used for value parsing";
569 w << "int success = false;";
570 w.skip_line();
571 }
572
573 // Write the argument handler for each argument
574 for (int i = 0; i < f.arguments.size(); i++)
575 {
576 export_argument_handler(w, i, f.arguments[i]);
577 }
578
579 export_consumer_function_value_handler(w, f);
580
581 w.skip_line();
582
583 w << "return true;";
584
585 w.unindent();
586 w << "}";
587 w.skip_line();
588}

590void export_argument_handler(Cpp_File_Writer &w, size_t i, const Argument &arg)
591{
592 // Create variable
593 w << std::format("// {} argument {}: '{} {}'", arg.has_default_value ? "Optional" : "Required",
594 i, value_type_to_cpp_type(arg.type), arg.name);
595
596 if (arg.has_default_value)
597 {
598 w << std::format("{} arg_{} = {};", value_type_to_cpp_type(arg.type), arg.name,
599 value_to_string(arg.default_value));
600
601 // Check if replacement variable has been provided
602 w << std::format("if(args.size() > {})", i);
603 w << "{";
604 w.indent();
605
606 // If it's a string, then we don't need parsing.
607 if (arg.type == Value_Type::STRING)
608 {
609 w << "success = true;";
610 w << std::format("arg_{} = args[{}];", arg.name, i);
611 }
612 else
613 {
614 w << std::format("success = get_{}(args[{}], arg_{});",
615 value_type_to_readable_string(arg.type), i, arg.name);
616 }
617 }
618 else
619 {
620 w << std::format("{} arg_{};", value_type_to_cpp_type(arg.type), arg.name);
621 if (arg.type == Value_Type::STRING)
622 {
623 w << "success = true;";
624 w << std::format("arg_{} = args[{}];", arg.name, i);
625 }
626 else
627 {
628 w << std::format("success = get_{}(args[{}], arg_{});",
629 value_type_to_readable_string(arg.type), i, arg.name);
630 }
631 }
632
633 w << "if(!success)";
634 w << "{";
635 w.indent();
636
637 w << std::format("printf(\"Failed to parse argument {0} '{1}'. Attempted to parse a {2}, but "
638 "got string '%s'\\n\", args[{0}].c_str());", i, arg.name, value_type_to_cpp_type(arg.type));
639 w << "return 0;";
640
641 w.unindent();
642 w << "}";
643
644 if (arg.has_default_value)
645 {
646 w.unindent();
647 w << "}";
648 }
649 w.skip_line();
650}

673void export_initialization_function(Cpp_File_Writer &w,
674 const std::vector<Function_Decl> &functions, const Settings &settings)
675{
676 // Write the initialization function
677 w << "//////////////////////////////";
678 w << "// INITIALIZER //";
679 w << "//////////////////////////////";
680 w.skip_line();
681
682 w << std::format("void {}(Function_Map &out_functions)", settings.init_function_name);
683 w << "{";
684 w.indent();
685
686 for (const auto &f : functions)
687 {
688 w << std::format("out_functions[\"{0}\"] = "
689 "Function_Decl(\"{0}\", _function_{0}, {1}, {2}, {3},",
690 f.name, value_type_to_string(f.return_type), f.num_required_args, f.num_optional_args);
691 w.indent();
692 w << std::format("\"{}\", (size_t){},", f.file, f.line);
693 w << "// Arguments";
694 w << "{";
695 w.indent();
696 for (size_t i = 0; i < f.arguments.size(); i++)
697 {
698 const auto &arg = f.arguments[i];
699 bool last_iter = i == f.arguments.size() - 1;
700
701 if (arg.has_default_value)
702 {
703 w << std::format("Argument(\"{}\", {}, {}){}",
704 arg.name, value_type_to_string(arg.type), value_to_string(arg.default_value),
705 last_iter ? "" : ",");
706 }
707 else
708 {
709 w << std::format("Argument(\"{}\", {}){}",
710 arg.name, value_type_to_string(arg.type), last_iter ? "" : ",");
711 }
712 }
713 w.unindent();
714 w << "},";
715 w << std::format("\"{}\"", f.note);
716 w.unindent();
717 w << ");";
718 w.skip_line();
719
720 w.unindent();
721 w << "}";
722 w.skip_line();
723 }

726std::string function_call_string(const Function_Decl &func)
727{
728 std::stringstream ss;
729 ss << func.name << "(";
730 for (int i = 0; i < func.arguments.size(); i++)
731 {
732 const Argument &arg = func.arguments[i];
733 ss << "arg_" << arg.name;
734
735 if (i < func.arguments.size() - 1)

737 ss << ", ";
738 }

884{

888 std::cerr << "[ERROR] Couldn't find file '" << path << "'! Terminating.\n";

890 }

895}
