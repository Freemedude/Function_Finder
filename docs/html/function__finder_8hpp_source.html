342inline size_t get_quoted_string(std::string_view source, std::string &out_result)
343{
344 size_t length = 0;
345 if (source[0] != '\"')
346 {
347 std::cerr << "[ERROR] get_quoted_string() takes a string that start with a quote (after whitespace)" << std::endl;
348 return 0;
349 }
350 length++;
351
352 while (source[length] && source[length] != '\"')
353 {
354 length++;
355 }
356
357 if (source[length] != '\"')
358 {
359 // We must terminate on a quote if the string is quoted!
360 return 0;
361 }
362
363 // Add the terminating quote into the length
364 length++;
365
366 out_result = std::string(source.begin() + 1, source.begin() + length - 1);
367
368 return length;
369}

431inline size_t get_double(std::string_view source, double &out_result)
432{
433 size_t max_length = source.size();
434 size_t total_length = get_word_length(source);
435 size_t length = 0;
436 bool had_digits = false;
437
438 if (length < max_length && (source[0] == '-' || source[0] == '+'))
439 {
440 length++;
441 }
442
443 while (length < max_length && std::isdigit(source[length]))
444 {
445 length++;
446 had_digits = true;
447 }
448
449 // Handle comma separation
450 if (length < max_length && source[length] == '.')
451 {
452 length++;
453 }
454
455 while (length < max_length && std::isdigit(source[length]))
456 {
457 length++;
458 had_digits = true;
459 }
460
461 // We must have SOME digits. Either '.1' or '1.' is allowed, but not just '.'
462 if (!had_digits)
463 {
464 return false;
465 }
466
467 if (length == 0 || length != total_length)
468 {
469 // We must have used all tokens here!
470 return false;
471 }
472
473 out_result = std::atof(source.data());
474
475 return length;
476}
